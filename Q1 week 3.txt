#include<iostream>
using namespace std;
void insertion(int a[],int n)
{
    int i,j,c=0,s=0;
    for(i=1;i<n;i++)
    {
        j=i-1;
        int temp=a[i];
        while(j>=0)
        {   
            c++;
            if(a[j]>temp)
            {
                a[j+1]=a[j];
                a[j]=temp;
                s++;
            }
            j--;
        }
    }
    for(i=0;i<n;i++)
        cout<<a[i]<<"  ";
    cout<<endl;
    cout<<"shifts="<<s<<endl;
    cout<<"comparisons="<<c<<endl;
}
int main()
{
    int t;
    cout<<"enter number of test cases";
    cin>>t;
    while(t--)
    {
        int n;
        cout<<"enter size";
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++)
            cin>>a[i];
        insertion(a,n);
    }
}
